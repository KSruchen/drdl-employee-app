-- Create tables
CREATE TABLE IF NOT EXISTS directorates (
  directorate_id INTEGER PRIMARY KEY,
  name VARCHAR(50)
);

CREATE TABLE IF NOT EXISTS departments (
  department_id INTEGER PRIMARY KEY,
  directorate_id INTEGER,
  name VARCHAR(50),
  CONSTRAINT fk_directorate FOREIGN KEY (directorate_id) REFERENCES directorates(directorate_id)
);

CREATE TABLE IF NOT EXISTS employees (
  emp_id INTEGER PRIMARY KEY,
  name VARCHAR(100),
  department_id INTEGER,
  email VARCHAR(100),
  CONSTRAINT fk_department FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

CREATE TABLE IF NOT EXISTS directors (
  director_id INTEGER PRIMARY KEY,
  name VARCHAR(100),
  role VARCHAR(50),
  directorate_id INTEGER NULL,
  start_date DATE,
  end_date DATE
);

CREATE TABLE IF NOT EXISTS department_heads (
  head_id INTEGER PRIMARY KEY,
  emp_id INTEGER,
  department_id INTEGER,
  start_date DATE,
  end_date DATE,
  CONSTRAINT fk_head_emp FOREIGN KEY (emp_id) REFERENCES employees(emp_id),
  CONSTRAINT fk_head_dept FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

CREATE TABLE IF NOT EXISTS employee_history (
  history_id INTEGER PRIMARY KEY,
  emp_id INTEGER,
  role VARCHAR(100),
  department_id INTEGER,
  start_date DATE,
  end_date DATE,
  CONSTRAINT fk_hist_emp FOREIGN KEY (emp_id) REFERENCES employees(emp_id),
  CONSTRAINT fk_hist_dept FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

-- Create View for easier employee info retrieval
CREATE OR REPLACE VIEW employee_details_vw AS
SELECT 
    e.emp_id AS "Employee ID",
    e.name AS "Employee Name",
    eh.role AS "Designation",
    d.name AS "Director",
    dirc.name AS "Directorate",
    dep.name AS "Division",
    eh.start_date AS "Joining Date",
    eh.end_date AS "End Date"
FROM 
    employees e
    LEFT JOIN employee_history eh ON e.emp_id = eh.emp_id AND eh.end_date IS NULL
    LEFT JOIN departments dep ON e.department_id = dep.department_id
    LEFT JOIN directors d ON dep.directorate_id = d.directorate_id AND d.end_date IS NULL
    LEFT JOIN directorates dirc ON dep.directorate_id = dirc.directorate_id;
